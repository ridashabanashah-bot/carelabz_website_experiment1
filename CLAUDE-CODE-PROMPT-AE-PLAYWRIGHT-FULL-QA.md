## What This Prompt Does

1. **Crawls every AE URL** — discovers all pages by following links from the homepage (BFS), plus explicitly hits every service page and blog post from Strapi
2. **Screenshots every page** — full-page screenshots saved to `tests/screenshots/ae/`
3. **Checks every page for content problems:**
   - Ellipses (`...` or `…`) that indicate truncated/missing content
   - Stray asterisks (`*`, `**`, `***`) that are markdown artifacts
   - Missing or cut-off letters/words (content that ends mid-sentence)
   - Broken formatting (visible HTML tags, markdown headers `##`, encoded entities)
   - Empty sections (headings with no content below them)
4. **Logs every issue** with the page URL, the problematic text, and where it appears
5. **Fixes every issue in Strapi CMS** via PUT API — for ellipses specifically, it fetches the original content from the WordPress source page (www.carelabz.com) and restores the full text
6. **Re-runs verification** after fixes to confirm everything is clean

**CRITICAL: Do NOT change any of the actual content/wording.** Only fix formatting artifacts (remove `**`, `...`, stray `#`, encoding issues). When restoring truncated text (ellipses), use the original WordPress content as the source of truth.

```python
#!/usr/bin/env python3
"""
AE Full Site QA — Phase 1: Discover all URLs
Combines Strapi API query + live site crawl for complete URL list.
"""

import os, sys, json, subprocess

try:
    import requests
except ImportError:
    subprocess.check_call([sys.executable, "-m", "pip", "install", "requests", "--break-system-packages", "-q"])
    import requests

STRAPI_BASE = "https://rational-cheese-8e8c4f80ea.strapiapp.com/api"
STRAPI_TOKEN = os.environ.get("STRAPI_API_TOKEN", "")
HEADERS = {"Authorization": f"Bearer {STRAPI_TOKEN}"}
REGION = "ae"
VERCEL_BASE = "https://carelabz-website-experiment1-ivory.vercel.app"

def get_all_ae_urls():
    """Get all AE URLs from Strapi."""
    urls = set()
    
    # Static pages
    urls.add("/ae/")
    urls.add("/ae/about/")
    urls.add("/ae/contact/")
    urls.add("/ae/services/")
    urls.add("/ae/blog/")
    urls.add("/ae/case-studies/")
    
    # Service pages
    r = requests.get(
        f"{STRAPI_BASE}/service-pages?filters[region][$eq]={REGION}&pagination[pageSize]=100&fields[0]=slug",
        headers=HEADERS, timeout=15,
    )
    for entry in r.json().get("data", []):
        slug = entry.get("slug", "")
        if slug:
            urls.add(f"/ae/services/{slug}/")
    
    # Blog posts
    r = requests.get(
        f"{STRAPI_BASE}/blog-posts?filters[region][$eq]={REGION}&pagination[pageSize]=100&fields[0]=slug",
        headers=HEADERS, timeout=15,
    )
    for entry in r.json().get("data", []):
        slug = entry.get("slug", "")
        if slug:
            urls.add(f"/ae/blog/{slug}/")
    
    # Case studies
    r = requests.get(
        f"{STRAPI_BASE}/case-studies?filters[region][$eq]={REGION}&pagination[pageSize]=100&fields[0]=slug",
        headers=HEADERS, timeout=15,
    )
    for entry in r.json().get("data", []):
        slug = entry.get("slug", "")
        if slug:
            urls.add(f"/ae/case-studies/{slug}/")
    
    return sorted(urls)

if __name__ == "__main__":
    urls = get_all_ae_urls()
    os.makedirs("data", exist_ok=True)
    with open("data/ae-all-urls.json", "w") as f:
        json.dump(urls, f, indent=2)
    print(f"Found {len(urls)} AE URLs")
    for u in urls[:10]:
        print(f"  {u}")
    if len(urls) > 10:
        print(f"  ... and {len(urls) - 10} more")
```

```ts
            url: urlPath,
            type: "MARKDOWN_HEADER",
            text: line.trim().slice(0, 150),

            url: urlPath,
            type: "HTML_ENTITY",

            context: "Unrendered HTML entity visible in text",

        bodyText.includes("[email protected]") ||
        fullHTML.includes("/cdn-cgi/l/email-protection")

          url: urlPath,
```

## Step 3: Fix All Issues in Strapi

Create `scripts/ae-fix-content-issues.py` — reads the QA report and fixes each issue in Strapi:

```python
#!/usr/bin/env python3
"""
AE Content Fix — reads QA report and fixes issues in Strapi CMS.
For ellipses: fetches original content from WordPress to restore full text.
For asterisks/formatting: cleans the Strapi field directly.
"""

import os, sys, json, re, subprocess, time

try:
    import requests
except ImportError:
    subprocess.check_call([sys.executable, "-m", "pip", "install", "requests", "--break-system-packages", "-q"])
    import requests

try:
    from bs4 import BeautifulSoup
    import markdownify
except ImportError:
    subprocess.check_call([
        sys.executable, "-m", "pip", "install",
        "beautifulsoup4", "markdownify", "--break-system-packages", "-q"
    ])
    from bs4 import BeautifulSoup
    import markdownify

STRAPI_BASE = "https://rational-cheese-8e8c4f80ea.strapiapp.com/api"
STRAPI_TOKEN = os.environ.get("STRAPI_API_TOKEN", "")
HEADERS_GET = {"Authorization": f"Bearer {STRAPI_TOKEN}"}
HEADERS_PUT = {
    "Authorization": f"Bearer {STRAPI_TOKEN}",
    "Content-Type": "application/json",
}
REGION = "ae"
WP_BASE = "https://www.carelabz.com"
DRY_RUN = "--dry-run" in sys.argv

# ─── URL → Strapi mapping ───

def url_to_strapi(url_path):
    """Convert a page URL to its Strapi content type and slug."""
    # /ae/services/some-slug-ae/ → ("service-pages", "some-slug-ae")
    # /ae/blog/some-slug-ae/ → ("blog-posts", "some-slug-ae")
    # /ae/case-studies/some-slug-ae/ → ("case-studies", "some-slug-ae")
    # /ae/about/ → ("about-pages", None)
    # /ae/contact/ → ("contact-pages", None)
    # /ae/ → ("home-pages", None)
    
    path = url_path.rstrip("/")
    
    if re.match(r"^/ae/services/(.+)$", path):
        slug = re.match(r"^/ae/services/(.+)$", path).group(1)
        return ("service-pages", slug)
    elif re.match(r"^/ae/blog/(.+)$", path):
        slug = re.match(r"^/ae/blog/(.+)$", path).group(1)
        return ("blog-posts", slug)
    elif re.match(r"^/ae/case-studies/(.+)$", path):
        slug = re.match(r"^/ae/case-studies/(.+)$", path).group(1)
        return ("case-studies", slug)
    elif path == "/ae/about":
        return ("about-pages", None)
    elif path == "/ae/contact":
        return ("contact-pages", None)
    elif path == "/ae" or path == "/ae/services" or path == "/ae/blog" or path == "/ae/case-studies":
        return ("home-pages", None)
    
    return (None, None)

def url_to_wordpress(url_path):
    """Convert a new site URL to the likely WordPress equivalent."""
    path = url_path.rstrip("/")
    
    # /ae/services/arc-flash-study-analysis-ae → carelabz.com/ae/arc-flash-study-analysis/
    # /ae/blog/some-post-ae → carelabz.com/ae/some-post/
    
    if "/services/" in path:
        slug = path.split("/services/")[-1]
        # Remove -ae suffix that was added during migration
        wp_slug = re.sub(r"-ae$", "", slug)
        # Try multiple WP URL patterns
        return [
            f"{WP_BASE}/ae/{wp_slug}/",
            f"{WP_BASE}/{wp_slug}/",
            f"{WP_BASE}/ae/services/{wp_slug}/",
        ]
    elif "/blog/" in path:
        slug = path.split("/blog/")[-1]
        wp_slug = re.sub(r"-ae$", "", slug)
        return [
            f"{WP_BASE}/ae/{wp_slug}/",
            f"{WP_BASE}/{wp_slug}/",
            f"{WP_BASE}/ae/blog/{wp_slug}/",
        ]
    
    return []

# ─── WordPress content extraction ───

def fetch_wordpress_content(wp_urls):
    """Try to fetch content from WordPress. Returns body text or None."""
    for url in wp_urls:
        try:
            r = requests.get(url, timeout=15, headers={
                "User-Agent": "Mozilla/5.0 (compatible; CareLabsMigration/1.0)"
            })
            if r.status_code == 200:
                soup = BeautifulSoup(r.text, "html.parser")
                
                # Find the main content area (WordPress themes vary)
                content = None
                for selector in [
                    "article .entry-content",
                    ".post-content",
                    ".entry-content",
                    "article",
                    ".content-area",
                    "main",
                ]:
                    content = soup.select_one(selector)
                    if content:
                        break
                
                if content:
                    # Remove scripts, styles, nav, footer
                    for tag in content.find_all(["script", "style", "nav", "footer", "aside"]):
                        tag.decompose()
                    
                    # Convert to markdown
                    md = markdownify.markdownify(str(content), heading_style="ATX")
                    
                    # Clean up
                    md = re.sub(r"\n{3,}", "\n\n", md)
                    md = md.strip()
                    
                    if len(md) > 100:
                        print(f"    ✓ Fetched WP content from {url} ({len(md)} chars)")
                        return md
        except Exception as e:
            continue
    
    return None

# ─── Strapi operations ───

def get_strapi_entry(content_type, slug=None):
    """Fetch a Strapi entry by content type and optional slug."""
    if slug:
        url = f"{STRAPI_BASE}/{content_type}?filters[slug][$eq]={slug}&filters[region][$eq]={REGION}&populate=*"
    else:
        url = f"{STRAPI_BASE}/{content_type}?filters[region][$eq]={REGION}&populate=*"
    
    r = requests.get(url, headers=HEADERS_GET, timeout=15)
    data = r.json().get("data", [])
    return data[0] if data else None

def update_strapi_entry(content_type, doc_id, update_data):
    """PUT update to Strapi."""
    if DRY_RUN:
        print(f"    [DRY RUN] Would update {content_type}/{doc_id}: {list(update_data.keys())}")
        return True
    
    r = requests.put(
        f"{STRAPI_BASE}/{content_type}/{doc_id}",
        headers=HEADERS_PUT,
        json={"data": update_data},
        timeout=15,
    )
    if r.status_code == 200:
        print(f"    ✓ Updated {content_type}/{doc_id}")
        return True
    else:
        print(f"    ✗ Failed {content_type}/{doc_id}: {r.status_code} — {r.text[:200]}")
        return False

# ─── Content cleaning ───

def clean_field(text):
    """Clean a text field of formatting artifacts. Returns (cleaned, changes)."""
    if not text or not isinstance(text, str):
        return text, []
    
    original = text
    changes = []
    
    # 1. Remove stray asterisks (markdown bold/italic that weren't rendered)
    # Empty bold: ** ** or ****
    cleaned = re.sub(r"\*{2}\s*\*{2}", "", text)
    if cleaned != text:
        changes.append("Removed empty bold markers")
        text = cleaned
    
    # Orphaned ** (single pair with no match on same line)
    lines = text.split("\n")
    fixed_lines = []
    for line in lines:
        count = line.count("**")
        if count == 1:
            fixed_line = line.replace("**", "")
            if fixed_line != line:
                changes.append(f"Removed orphaned **: {line[:60]}")
                line = fixed_line
        fixed_lines.append(line)
    text = "\n".join(fixed_lines)
    
    # Stray single * between words (not list items)
    cleaned = re.sub(r"(?<=\w)\*(?=\w)", "", text)
    if cleaned != text:
        changes.append("Removed stray * between words")
        text = cleaned
    
    # 2. Remove stray markdown headers mid-text
    cleaned = re.sub(r"(?<=\w) #{2,3} (?=\w)", " — ", text)
    if cleaned != text:
        changes.append("Replaced mid-line ## with dash")
        text = cleaned
    
    # 3. HTML entities
    entities = {
        "&amp;": "&", "&nbsp;": " ", "&rsquo;": "'", "&lsquo;": "'",
        "&rdquo;": '"', "&ldquo;": '"', "&mdash;": "—", "&ndash;": "–",
        "&hellip;": "…", "&bull;": "•", "&#8217;": "'", "&#8220;": '"',
        "&#8221;": '"', "&#160;": " ",
    }
    for entity, replacement in entities.items():
        if entity in text:
            text = text.replace(entity, replacement)
            changes.append(f"Decoded {entity}")
    
    # 4. Cloudflare artifacts
    if "/cdn-cgi/l/email-protection" in text:
        text = re.sub(r"\[email\s*protected\]\(/cdn-cgi/l/email-protection[^)]*\)", "info@carelabz.com", text)
        text = re.sub(r"/cdn-cgi/l/email-protection[^\s\"'<)]*", "mailto:info@carelabz.com", text)
        changes.append("Replaced Cloudflare artifact")
    
    # 5. null prefix
    if "nullinfo@" in text:
        text = text.replace("nullinfo@carelabz.com", "info@carelabz.com")
        changes.append("Fixed null email prefix")
    
    # 6. Multiple spaces
    cleaned = re.sub(r"[^\S\n]{2,}", " ", text)
    if cleaned != text:
        changes.append("Collapsed multiple spaces")
        text = cleaned
    
    # 7. Excessive blank lines
    cleaned = re.sub(r"\n{3,}", "\n\n", text)
    if cleaned != text:
        changes.append("Reduced excessive blank lines")
        text = cleaned
    
    # 8. Trim
    text = text.strip()
    
    return text, changes

def fix_ellipses_from_wordpress(entry, content_type, url_path):
    """
    For fields containing ellipses, fetch the original WordPress page
    and replace the truncated text with the full original.
    """
    changes = []
    update = {}
    
    # Text fields to check
    text_fields = [
        "body", "definitionalLede", "heroSubtext", "heroHeadline",
        "missionBody", "challenge", "solution", "excerpt",
        "ctaBannerBody", "ctaBannerHeading", "safetyBody",
        "metaDescription",
    ]
    
    # Check which fields have ellipses
    fields_with_ellipsis = []
    for field in text_fields:
        val = entry.get(field, "")
        if isinstance(val, str) and ("..." in val or "…" in val):
            fields_with_ellipsis.append(field)
    
    if not fields_with_ellipsis:
        return {}, []
    
    print(f"  Fields with ellipses: {fields_with_ellipsis}")
    
    # Fetch original WordPress content
    wp_urls = url_to_wordpress(url_path)
    wp_content = fetch_wordpress_content(wp_urls) if wp_urls else None
    
    if wp_content:
        # For the body field specifically, replace with WP content if it's substantially longer
        if "body" in fields_with_ellipsis:
            current_body = entry.get("body", "")
            # Only replace if WP content is significantly longer (not just a few chars)
            if len(wp_content) > len(current_body) * 1.2:
                # Clean the WP content too
                cleaned_wp, _ = clean_field(wp_content)
                # Trim to 5000 chars for service pages, 8000 for blog posts
                max_len = 8000 if "blog" in content_type else 5000
                if len(cleaned_wp) > max_len:
                    # Find a sentence break near the limit
                    cutoff = cleaned_wp[:max_len].rfind(". ")
                    if cutoff > max_len * 0.8:
                        cleaned_wp = cleaned_wp[:cutoff + 1]
                    else:
                        cleaned_wp = cleaned_wp[:max_len]
                
                update["body"] = cleaned_wp
                changes.append(f"Restored body from WordPress ({len(current_body)} → {len(cleaned_wp)} chars)")
                fields_with_ellipsis.remove("body")
        
        # For other fields with ellipses, try to find matching content in WP page
        for field in fields_with_ellipsis:
            current = entry.get(field, "")
            if not current:
                continue
            
            # Get the text before the ellipsis
            before_ellipsis = re.split(r"\.{3}|…", current)[0].strip()
            if len(before_ellipsis) < 10:
                continue
            
            # Search for this text prefix in the WP content
            search_text = before_ellipsis[-50:]  # Last 50 chars before ellipsis
            if search_text in wp_content:
                # Find the full sentence/paragraph in WP content
                idx = wp_content.index(search_text)
                # Grab from there to the next paragraph break or end
                rest = wp_content[idx + len(search_text):]
                end_idx = rest.find("\n\n")
                if end_idx == -1:
                    end_idx = min(len(rest), 500)
                
                full_text = before_ellipsis + rest[:end_idx].strip()
                cleaned_full, _ = clean_field(full_text)
                
                # Respect field length limits
                max_field_len = 160 if "meta" in field.lower() else 500
                if len(cleaned_full) > max_field_len:
                    cutoff = cleaned_full[:max_field_len].rfind(". ")
                    if cutoff > max_field_len * 0.7:
                        cleaned_full = cleaned_full[:cutoff + 1]
                
                update[field] = cleaned_full
                changes.append(f"Restored {field} from WordPress")
    else:
        # No WP content available — just remove the ellipses
        # (replace "..." with a period if it ends a sentence, otherwise just remove)
        for field in fields_with_ellipsis:
            current = entry.get(field, "")
            # Replace trailing "..." with "."
            fixed = re.sub(r"\s*\.{3}\s*$", ".", current)
            # Replace mid-text "..." with " — "
            fixed = re.sub(r"\s*\.{3}\s*", " — ", fixed)
            # Same for unicode ellipsis
            fixed = re.sub(r"\s*…\s*$", ".", fixed)
            fixed = re.sub(r"\s*…\s*", " — ", fixed)
            
            if fixed != current:
                update[field] = fixed
                changes.append(f"Cleaned ellipsis in {field} (no WP source available)")
    
    return update, changes

# ─── Main ───

def main():
    if DRY_RUN:
        print("=" * 60)
        print("DRY RUN — no changes will be saved")
        print("=" * 60)
    
    # Load QA report
    report_path = "data/ae-qa-report.json"
    if not os.path.exists(report_path):
        print(f"No QA report found at {report_path}")
        print("Run the Playwright QA suite first: npx playwright test tests/ae-full-qa.spec.ts")
        sys.exit(1)
    
    with open(report_path) as f:
        issues = json.load(f)
    
    print(f"Loaded {len(issues)} issues from QA report\n")
    
    # Group issues by URL
    by_url = {}
    for issue in issues:
        url = issue["url"]
        by_url.setdefault(url, []).append(issue)
    
    total_fixed = 0
    total_skipped = 0
    
    for url_path, page_issues in sorted(by_url.items()):
        print(f"\n{'='*60}")
        print(f"Page: {url_path} ({len(page_issues)} issues)")
        print(f"{'='*60}")
        
        content_type, slug = url_to_strapi(url_path)
        if not content_type:
            print(f"  ⚠ Cannot map to Strapi content type — skipping")
            total_skipped += len(page_issues)
            continue
        
        # Fetch the Strapi entry
        entry = get_strapi_entry(content_type, slug)
        if not entry:
            print(f"  ⚠ No Strapi entry found for {content_type}/{slug} — skipping")
            total_skipped += len(page_issues)
            continue
        
        doc_id = entry.get("documentId", entry.get("id"))
        issue_types = set(i["type"] for i in page_issues)
        print(f"  Issue types: {issue_types}")
        
        update = {}
        all_changes = []
        
        # Fix formatting issues across all text fields
        for field in ["body", "definitionalLede", "heroSubtext", "heroHeadline",
                       "missionBody", "challenge", "solution", "excerpt",
                       "ctaBannerBody", "ctaBannerHeading", "safetyBody",
                       "safetyHeading", "safetyEyebrow", "processHeading",
                       "featuresHeading", "faqSectionHeading",
                       "metaTitle", "metaDescription", "title", "eyebrow"]:
            val = entry.get(field, "")
            if not isinstance(val, str) or not val:
                continue
            
            cleaned, changes = clean_field(val)
            if changes:
                update[field] = cleaned
                all_changes.extend([f"{field}: {c}" for c in changes])
        
        # Fix nested arrays (faqs, features, processSteps, etc.)
        for array_field in ["faqs", "features", "processSteps", "safetyBullets",
                            "industries", "insights", "values", "stats",
                            "certifications", "results", "services", "trustBadges"]:
            arr = entry.get(array_field)
            if not arr or not isinstance(arr, list):
                continue
            
            updated_arr = []
            arr_changed = False
            for item in arr:
                if not isinstance(item, dict):
                    updated_arr.append(item)
                    continue
                new_item = dict(item)
                new_item.pop("id", None)  # Strip component id for Strapi v5
                for k, v in item.items():
                    if isinstance(v, str) and v:
                        cleaned, changes = clean_field(v)
                        if changes:
                            new_item[k] = cleaned
                            all_changes.extend([f"{array_field}[].{k}: {c}" for c in changes])
                            arr_changed = True
                updated_arr.append(new_item)
            
            if arr_changed:
                update[array_field] = updated_arr
        
        # Handle ellipses specifically — try to restore from WordPress
        if "ELLIPSIS" in issue_types:
            wp_update, wp_changes = fix_ellipses_from_wordpress(entry, content_type, url_path)
            update.update(wp_update)
            all_changes.extend(wp_changes)
        
        # Apply updates
        if update:
            print(f"  Fixes to apply ({len(all_changes)}):")
            for c in all_changes[:15]:
                print(f"    - {c}")
            if len(all_changes) > 15:
                print(f"    ... and {len(all_changes) - 15} more")
            
            success = update_strapi_entry(content_type, doc_id, update)
            if success:
                total_fixed += len(all_changes)
            time.sleep(0.3)
        else:
            print(f"  No Strapi fixes needed (issues may be in code/layout)")
            total_skipped += len(page_issues)
    
    print(f"\n{'='*60}")
    print(f"DONE")
    print(f"{'='*60}")
    print(f"Total fixes applied: {total_fixed}")
    print(f"Issues skipped (not fixable in Strapi): {total_skipped}")

if __name__ == "__main__":
    main()
```

## Step 5: Post-Fix Verification Suite

After all fixes are applied, re-run a strict verification that enforces ZERO issues:

```ts
// tests/ae-post-fix-verify.spec.ts
import { test, expect } from "@playwright/test";
import * as fs from "fs";
import * as path from "path";

const BASE = "https://carelabz-website-experiment1-ivory.vercel.app";

// Load all URLs
let AE_URLS: string[] = [];
try {
  AE_URLS = JSON.parse(
    fs.readFileSync(path.join(__dirname, "..", "data", "ae-all-urls.json"), "utf-8")
  );
} catch {
  AE_URLS = ["/ae/", "/ae/services/", "/ae/blog/", "/ae/about/", "/ae/contact/"];
}

test.describe("AE Post-Fix Verification — Zero Tolerance", () => {
  test.setTimeout(600000);

  for (const urlPath of AE_URLS) {
    test(`Clean: ${urlPath}`, async ({ page }) => {
      const res = await page.goto(`${BASE}${urlPath}`, {
        waitUntil: "domcontentloaded",
        timeout: 20000,
      });

      // Must not be 404/500
      expect(res?.status(), `${urlPath} returned ${res?.status()}`).toBeLessThan(400);

      const bodyText = await page.locator("main").innerText().catch(() => "");
      const fullHTML = await page.content();

      // No stray asterisks in visible text
      const asteriskLines = bodyText
        .split("\n")
        .filter((line) => /(?:^|\s)\*{1,3}[^*]|[^*]\*{1,3}(?:\s|$)/.test(line.trim()))
        .filter((line) => !/^[-•]\s/.test(line.trim()));
      expect(
        asteriskLines.length,
        `Stray asterisks on ${urlPath}: ${asteriskLines[0]?.slice(0, 80)}`
      ).toBe(0);

      // No markdown headers visible as text
      const mdHeaders = bodyText
        .split("\n")
        .filter((line) => /^\s*#{1,4}\s+\w/.test(line.trim()));
      expect(mdHeaders.length, `Markdown headers on ${urlPath}`).toBe(0);

      // No unrendered HTML entities
      expect(bodyText).not.toMatch(
        /&(?:amp|nbsp|rsquo|ldquo|rdquo|hellip|mdash|ndash|#\d{3,5});/
      );

      // No Cloudflare artifacts
      expect(fullHTML).not.toContain("/cdn-cgi/l/email-protection");
      expect(bodyText).not.toContain("[email protected]");

      // No null emails
      expect(fullHTML).not.toContain("nullinfo@");
      expect(fullHTML).not.toContain("nullmailto:");

      // No dead carelabs.me links
      const deadLinks = await page.locator('a[href*="carelabs.me"]').count();
      expect(deadLinks, `Dead carelabs.me links on ${urlPath}`).toBe(0);
    });
  }
});
```
